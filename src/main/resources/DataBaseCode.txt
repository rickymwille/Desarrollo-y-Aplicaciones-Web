CREATE DATABASE StoriesForEveryone;
USE StoriesForEveryone;

CREATE TABLE usuarios (
  id int NOT NULL AUTO_INCREMENT,
  username varchar(45) NOT NULL,
  nombre varchar(45) NOT NULL,
  apellido varchar(45) NOT NULL,
  email varchar(75) NOT NULL,
  contrasena varchar(45) NOT NULL,
  activo int NOT NULL,
  roles VARCHAR(45) NOT NULL,
  permissions VARCHAR(45) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE blogs (
  id int NOT NULL AUTO_INCREMENT,
  TituloBlog varchar(50) NOT NULL,
  Fecha date NOT NULL,
  Entry varchar(500) NOT NULL,
  id_usuario int NOT NULL,
  PRIMARY KEY (id),
  FOREIGN KEY (id_usuario) REFERENCES usuarios(id)
);

CREATE TABLE aboutus_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE blog_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE contactus_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE index_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE login_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE profile_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE rewards_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);

CREATE TABLE signup_comments (
  id int NOT NULL AUTO_INCREMENT,
  Entry varchar(500) NOT NULL,
  PRIMARY KEY (id)
);
